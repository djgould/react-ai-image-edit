# Usage

```javascript
import AIEdit from "@djgould/react-ai-image-edit";

const App = () => {
  return (
    <AIEdit
      src="https://example.com/image.jpg"
      stabilityApiKey="sk-yourkey"
      width={500}
      height={500}
    />
  );
};
```
